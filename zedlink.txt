async def resolve_user_identifier(identifier, chat_id=None):
    try:
        identifier = str(identifier).strip()
        
        if identifier.isdigit():
            user_id = int(identifier)
            try:
                await client.get_entity(user_id)
                return user_id
            except:
                return None
        
        if identifier.startswith('@'):
            username = identifier[1:]
            try:
                user = await client.get_entity(username)
                return user.id
            except:
                return None
        
        if 't.me/' in identifier or 'telegram.me/' in identifier:
            try:
                user = await client.get_entity(identifier)
                return user.id
            except:
                return None
        
        try:
            user = await client.get_entity(identifier)
            return user.id
        except:
            return None
            
    except:
        return None

async def get_target_user(event, require_reply=False):
    target_user = None
    
    if event.reply_to_msg_id:
        reply_msg = await event.get_reply_message()
        if reply_msg:
            return reply_msg.sender_id
    
    if require_reply:
        return None
    
    parts = event.text.split()
    for part in parts[1:]:
        target_user = await resolve_user_identifier(part, event.chat_id)
        if target_user:
            return target_user
    
    return target_user

async def get_target_user_from_text(text):
    try:
        parts = text.split(maxsplit=2)
        if len(parts) > 1:
            identifier = parts[1].strip()
            return await resolve_user_identifier(identifier)
    except:
        pass
    return None

def parse_duration(text):
    text = text.replace('ساعت', 'hour').replace('دقیقه', 'minute').replace('ثانیه', 'second').replace('روز', 'day')
    
    day_match = re.search(r'(\d+)\s*day', text)
    hour_match = re.search(r'(\d+)\s*hour', text)
    minute_match = re.search(r'(\d+)\s*minute', text)
    second_match = re.search(r'(\d+)\s*second', text)
    
    seconds = 0
    if day_match:
        seconds += int(day_match.group(1)) * 86400
    if hour_match:
        seconds += int(hour_match.group(1)) * 3600
    if minute_match:
        seconds += int(minute_match.group(1)) * 60
    if second_match:
        seconds += int(second_match.group(1))
    
    return seconds

def format_time(seconds):
    days = seconds // 86400
    hours = (seconds % 86400) // 3600
    minutes = (seconds % 3600) // 60
    secs = seconds % 60
    
    parts = []
    if days > 0:
        parts.append(f"{days} روز")
    if hours > 0:
        parts.append(f"{hours} ساعت")
    if minutes > 0:
        parts.append(f"{minutes} دقیقه")
    if secs > 0:
        parts.append(f"{secs} ثانیه")
    
    return "، ".join(parts)

@client.on(events.CallbackQuery(pattern=r'report_(ban|mute|tempmute|warn|delete|ignore)_'))
async def handle_report_action(event):
    data = event.data.decode('utf-8')
    chat_id = event.sender_id
    user_id = event.sender_id
    
    parts = data.split('_')
    action = parts[1]
    target_chat_id = int(parts[2])
    target_user_id = int(parts[3])
    group_message_id = int(parts[4])
    forwarded_message_id = int(parts[5]) if len(parts) > 5 else None
    
    if action == "tempmute":
        await event.edit("چقدر میخواهید کاربر فلان سکوت زماندار شود؟\n(به عنوان مثال 30 ثانیه یا 30 دقیقه یا 30 ساعت یا 30 روز میتوانید ارسال کنید)")
        
        report_states[(user_id, target_chat_id, target_user_id, group_message_id)] = {
            'action': 'tempmute',
            'forwarded_message_id': forwarded_message_id
        }
        return
    
    try:
        if action == "ban":
            if target_chat_id not in banned_users:
                banned_users[target_chat_id] = {}
            
            banned_users[target_chat_id][target_user_id] = user_id
            await client.edit_permissions(target_chat_id, target_user_id, view_messages=False)
            
            buttons = [[Button.inline("حذف بن", f"unban_{target_chat_id}_{target_user_id}")]]
            await event.edit("بن", buttons=buttons)
            
        elif action == "mute":
            if target_chat_id not in muted_users:
                muted_users[target_chat_id] = {}
            
            muted_users[target_chat_id][target_user_id] = user_id
            await client.edit_permissions(target_chat_id, target_user_id, send_messages=False)
            
            await event.edit("سکوت")
            
        elif action == "warn":
            if target_chat_id not in warning_users:
                warning_users[target_chat_id] = {}
            
            if target_user_id not in warning_users[target_chat_id]:
                warning_users[target_chat_id][target_user_id] = 0
            
            warning_users[target_chat_id][target_user_id] += 1
            warnings_count = warning_users[target_chat_id][target_user_id]
            
            user_mention = await get_user_mention(target_user_id)
            await event.edit(f"اخطار به {user_mention} (تعداد اخطارها: {warnings_count}/3)")
            
            if warnings_count >= 3:
                await handle_warning_action(target_chat_id, target_user_id, warnings_count)
                warning_users[target_chat_id][target_user_id] = 0
                
        elif action == "delete":
            try:
                await client.delete_messages(target_chat_id, forwarded_message_id)
                await event.edit("حذف پیام")
            except:
                await event.edit("حذف پیام")
                
        elif action == "ignore":
            admin_mention = await get_user_mention(user_id)
            target_mention = await get_user_mention(target_user_id)
            await event.edit(f"ربات به دستور ادمین {admin_mention} هیچ اقدامی روی کاربر {target_mention} انجام نمیدهد (کاربر میوخورد)")
        
        await notify_group_action(target_chat_id, user_id, target_user_id, action, forwarded_message_id)
        
    except:
        await event.answer()

@client.on(events.NewMessage(pattern=r'^حذف (\d+)$'))
async def delete_multiple_messages(event):
    if not await is_owner_or_admin(event):
        return
    
    try:
        text = event.text
        number = int(text.split()[1])
        
        if number <= 0 or number > 100:
            return
        
        try:
            await client.delete_messages(event.chat_id, event.id)
        except:
            pass
        
        messages_to_delete = []
        count = 0
        
        async for message in client.iter_messages(event.chat_id, limit=number + 10):
            if message.sender_id == (await client.get_me()).id:
                continue
            
            if count >= number:
                break
            
            if await should_delete_message(message):
                messages_to_delete.append(message.id)
                count += 1
        
        if messages_to_delete:
            await client.delete_messages(event.chat_id, messages_to_delete)
        
        if count > 0:
            result_msg = await client.send_message(event.chat_id, f"**تعداد {count} پیام با موفقیت حذف شد!**")
            
            await asyncio.sleep(3)
            await client.delete_messages(event.chat_id, result_msg.id)
        
    except:
        pass

        return

    match = re.match(r'^قفل گروه (\d+) (ساعت|دقیقه|ثانیه)$', event.text)
    
    if match:
        number = int(match.group(1))
        unit = match.group(2)
        
        if unit == 'ساعت':
            duration_seconds = number * 3600
        elif unit == 'دقیقه':
            duration_seconds = number * 60
        else:
            duration_seconds = number
        
        group_lock_status[chat_id] = True
        
        end_time = time.time() + duration_seconds
        group_lock_end_time[chat_id] = end_time
        
        if chat_id in group_lock_timers:
            group_lock_timers[chat_id].cancel()
        
        task = asyncio.create_task(unlock_after_duration(chat_id, duration_seconds))
        group_lock_timers[chat_id] = task
        
        duration_text = format_time(duration_seconds)
        await event.reply(f"**قفل گروه به مدت {duration_text} فعال شد!**")
        
        if chat_id in admin_reports and admin_reports[chat_id]:
            admin_name = await get_user_name(event.sender_id)
            report_text = f"**{admin_name} قفل گروه را به مدت {duration_text} فعال کرد!**"
            await send_admin_report(chat_id, report_text)

    await event.reply("**ابتدا چیزی که میخواهید ربات بهش پاسخ دهد را ارسال کنید**\n\nبرای لغو، کلمه \"لغو\" را ارسال کنید.")

@client.on(events.NewMessage)
async def handle_all_messages(event):
    if not event.is_group:
        return
    
    chat_id = event.chat_id
    sender_id = event.sender_id
    
    if event.sender_id == (await client.get_me()).id:
        return
    
    state_key = (chat_id, sender_id)
    
    if state_key in user_states:
        state_data = user_states[state_key]
        
        if state_data['state'] == 'waiting_for_trigger':
            if event.text and event.text.strip() == 'لغو':
                del user_states[state_key]
                await event.reply("**عملیات تنظیم پاسخ لغو شد!**")
                return
            
            if event.text and event.text.strip() == 'تنظیم پاسخ':
                await event.reply("**لطفا متنی غیر از \"تنظیم پاسخ\" را ارسال کنید!**")
                return
            
            user_states[state_key] = {
                'state': 'waiting_for_response',
                'trigger': event.message
            }
            
            trigger_text = event.text[:30] + "..." if event.text and len(event.text) > 30 else (event.text if event.text else "این پیام")
            await event.reply(f"**حالا چیزی که میخواهید وقتی کسی \"{trigger_text}\" رو گفت پاسخ دهد را ارسال کنید**\n\nبرای لغو، کلمه \"لغو\" را ارسال کنید.")
            return
        
        elif state_data['state'] == 'waiting_for_response':
            if event.text and event.text.strip() == 'لغو':
                del user_states[state_key]
                await event.reply("**عملیات تنظیم پاسخ لغو شد!**")
                return
            
            trigger_content = state_data['trigger']
            response_content = event.message
            user_states[state_key] = {
                'state': 'waiting_for_access_level',
                'trigger': trigger_content,
                'response': response_content
            }
            
            buttons = [
                [Button.inline("مالک گروه", "owner_only")],
                [Button.inline("ادمین گروه", "admins_only")],
                [Button.inline("همه", "everyone")],
                [Button.inline("لغو", "cancel_auto_reply")]
            ]
            
            trigger_text = trigger_content.text[:30] + "..." if trigger_content.text and len(trigger_content.text) > 30 else (trigger_content.text if trigger_content.text else "رسانه")
            
            await event.reply(
                f"**لطفا از پنل زیر سطح دسترسی را برای پاسخ به \"{trigger_text}\" انتخاب کنید:**",
                buttons=buttons
            )
            return
    
    if event.text and event.text.strip() == 'پینگ':
        msg = await event.reply("**درحال پینگ گرفتن، لطفا صبرکنید**")
        start_time = time.time()
        try:
            await msg.edit("**درحال محاسبه پینگ...**")
        except:
            pass
        end_time = time.time()
        
        ping_time = (end_time - start_time) * 1000
        
        if ping_time < 100:
            status = "عالی"
        elif ping_time < 300:
            status = "خوب"
        else:
            status = "بد"
        
        await msg.edit(f"**ضدلینک امیرولف آنلاین میباشد!**\n**پینگ :** `{ping_time:.0f}ms`\n**وضعیت ربات :** `{status}`")
        return
    
    if event.text and event.text.strip() == 'پیکربندی':
        if not await is_bot_owner(event):
            return
        
        buttons = [
            [Button.inline("تنظیم مجدد ادمین گروه", "reset_admins")],
            [Button.inline("بروزرسانی ادمینای گروه", "update_group_admins")],
            [Button.inline("حذف لیست مدیران ربات", "clear_bot_admins_list")],
            [Button.inline("رفرش ربات", "refresh_bot")],
            [Button.inline("تایید", "confirm_config"), Button.inline("لغو", "cancel_config")]
        ]
        await event.reply("**ربات مجددا بروز شد!**\n\nلطفا از منو زیر انتخاب کنید", buttons=buttons)
        return
    
    if event.text and event.text.strip() == 'گزارش وضعیت فعال':
        if not await is_bot_owner(event):
            return
        
        admin_reports[chat_id] = True
        await event.reply("**گزارش وضعیت ادمین های گروه با موفقیت فعال شد!**")
        return

@client.on(events.CallbackQuery(pattern=r'owner_only|admins_only|everyone|cancel_auto_reply'))
async def handle_auto_reply_callback(event):
    data = event.data.decode('utf-8')
    chat_id = event.chat_id
    user_id = event.sender_id
    
    state_key = (chat_id, user_id)
    
    if state_key in user_states and user_states[state_key]['state'] == 'waiting_for_access_level':
        if data == 'cancel_auto_reply':
            await event.edit("**عملیات لغو شد!**")
            del user_states[state_key]
            return
        
        state_data = user_states[state_key]
        
        if data == 'owner_only':
            access_text = 'مالک گروه'
        elif data == 'admins_only':
            access_text = 'ادمین گروه'
        else:
            access_text = 'همه'
        
        trigger_content = state_data['trigger']
        response_content = state_data['response']
        
        if chat_id not in auto_replies:
            auto_replies[chat_id] = []
        
        auto_replies[chat_id].append({
            'trigger': trigger_content,
            'response': response_content,
            'access_level': data
        })
        
        trigger_text = trigger_content.text[:30] + "..." if trigger_content.text and len(trigger_content.text) > 30 else (trigger_content.text if trigger_content.text else "رسانه")
        response_text = response_content.text[:30] + "..." if response_content.text and len(response_content.text) > 30 else (response_content.text if response_content.text else "رسانه")
        
        await event.edit(f"**پیام \"{response_text}\" در پاسخ به \"{trigger_text}\" تنظیم شد!**\n\n**سطح دسترسی برای پیام :** `{access_text}`")
        del user_states[state_key]

@client.on(events.CallbackQuery(pattern=r'reset_admins|update_group_admins|clear_bot_admins_list|refresh_bot|confirm_config|cancel_config'))
async def handle_config_callback(event):
    data = event.data.decode('utf-8')
    chat_id = event.chat_id
    
    if data == "reset_admins":
        try:
            admins = await client.get_participants(chat_id, filter=types.ChannelParticipantsAdmins)
            await event.answer(f"تعداد {len(admins)} ادمین شناسایی شدند")
        except:
            await event.answer()
    
    elif data == "update_group_admins":
        try:
            admins = await client.get_participants(chat_id, filter=types.ChannelParticipantsAdmins)
            count = 0
            
            if chat_id not in bot_admins:
                bot_admins[chat_id] = []
            
            for admin in admins:
                if admin.bot:
                    continue
                
                if admin.id not in bot_admins[chat_id]:
                    bot_admins[chat_id].append(admin.id)
                    count += 1
            
            await event.answer(f"تعداد {count} ادمین به لیست مدیران ربات اضافه شدند")
            
            if chat_id in admin_reports and admin_reports[chat_id]:
                report_text = f"**بروزرسانی ادمینای گروه انجام شد! تعداد {count} ادمین اضافه شدند.**"
                await send_admin_report(chat_id, report_text)
        
        except:
            await event.answer()
    
    elif data == "clear_bot_admins_list":
        if chat_id in bot_admins:
            bot_admins[chat_id] = []
            await event.answer("لیست مدیران ربات پاکسازی شد")
        else:
            await event.answer("لیست مدیران ربات خالی است")
    
    elif data == "refresh_bot":
        await event.edit("**ربات در حال رفرش میباشد، این عملیات 10 ثانیه طول میکشد**")
        await asyncio.sleep(10)
        
        try:
            admins = await client.get_participants(chat_id, filter=types.ChannelParticipantsAdmins)
            real_owner_count = 0
            for admin in admins:
                try:
                    participant = await client.get_permissions(chat_id, admin.id)
                    if participant.is_creator:
                        real_owner_count += 1
                except:
                    continue
            
            admin_count = len([a for a in admins if not a.bot])
        except:
            admin_count = 0
            real_owner_count = 0
        
        bot_admin_count = len(bot_admins.get(chat_id, []))
        
        await event.edit(f"**ربات با موفقیت رفرش شد!**\n**تعداد ادمین های گروه :** `{admin_count}`\n**تعداد مالک واقعی گروه :** `{real_owner_count}`\n**تعداد مدیران ربات :** `{bot_admin_count}`")
    
    elif data == "confirm_config":
        await event.edit("**تنظیمات تایید شد!**")
    
    elif data == "cancel_config":
        await event.edit("**عملیات لغو شد!**")

@client.on(events.NewMessage(pattern='^لیست پاسخ$'))
async def list_auto_replies(event):
    if not await is_owner_or_admin(event):
        return
    
    chat_id = event.chat_id
    
    if chat_id not in auto_replies or not auto_replies[chat_id]:
        await event.reply("**لیست پاسخ‌های خودکار خالی است!**")
        return
    
    message = "**لیست پاسخ‌های خودکار**\n\n"
    
    for i, reply in enumerate(auto_replies[chat_id], 1):
        trigger = reply['trigger']
        response = reply['response']
        access_level = reply['access_level']
        
        trigger_text = trigger.text[:20] + "..." if trigger.text and len(trigger.text) > 20 else (trigger.text if trigger.text else "رسانه")
        response_text = response.text[:20] + "..." if response.text and len(response.text) > 20 else (response.text if response.text else "رسانه")
        
        if access_level == 'owner_only':
            access_text = 'مالک گروه'
        elif access_level == 'admins_only':
            access_text = 'ادمین گروه'
        else:
            access_text = 'همه'
        
        message += f"**{i} -** پاسخ به \"{trigger_text}\" با \"{response_text}\" (سطح: `{access_text}`)\n"
    
    await event.reply(message)

@client.on(events.NewMessage(pattern='^حذف پاسخ$'))
async def delete_auto_reply(event):
    if not await is_owner_or_admin(event):
        return
    
    chat_id = event.chat_id
    
    if chat_id not in auto_replies or not auto_replies[chat_id]:
        await event.reply("**لیست پاسخ‌های خودکار خالی است!**")
        return
    
    if event.reply_to_msg_id:
        reply_msg = await event.get_reply_message()
        if reply_msg and reply_msg.text:
            trigger_text = reply_msg.text.strip()
            
            for i, reply in enumerate(auto_replies[chat_id]):
                if reply['trigger'].text and reply['trigger'].text.strip() == trigger_text:
                    del auto_replies[chat_id][i]
                    await event.reply(f"**پاسخ خودکار برای \"{trigger_text[:30]}\" حذف شد!**")
                    return
        
        await event.reply("**پاسخ خودکار با این متن یافت نشد!**")
    else:
        await event.reply("**لطفا روی پیامی که میخواهید پاسخ آن حذف شود ریپلای کنید!**")
